const parseItems = (id) =>
  getItems(id)
    .then((res) =>
      res.data.reduce((result, item) => {
        return { ...result, [item.item.name]: item.quantity }
      }, {})
    ).then((items) => {
      const initial = { Water:0, Food:0, Ammunition:0, Medication:0 };
      return { ...initial, ...items  }
    });
